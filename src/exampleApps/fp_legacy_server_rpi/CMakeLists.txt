project(fp_legacy_server_rpi)

cmake_minimum_required(VERSION 3.6)

#---------------------------------------------------------------------------------------
# Build target
#---------------------------------------------------------------------------------------

link_directories(
    ${CMAKE_SOURCE_DIR}/src/opend/dialog/fp_hanfun/include/lib_linux
    ${CMAKE_SOURCE_DIR}/src/opend/dspg/fp_hanfun/include/lib
)

file(GLOB SOURCES
    "${CMAKE_CURRENT_SOURCE_DIR}/fp_legacy_server.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/udp_server.c"
)

add_executable(${PROJECT_NAME} ${SOURCES})

#---------------------------------------------------------------------------------------
# Include
#---------------------------------------------------------------------------------------

target_include_directories(${PROJECT_NAME} PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
)


#---------------------------------------------------------------------------------------
# Link target
#---------------------------------------------------------------------------------------
target_link_libraries(${PROJECT_NAME}
    stdc++
    legacy_api
    pthread
    utils
)
